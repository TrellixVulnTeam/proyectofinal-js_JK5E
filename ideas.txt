porque si es par es - y si es impar o numero con coma es +


for(let i=0; i<modificarCantidad.length; i++){
			modificarCantidad[i].addEventListener("click",function(e){
				localArticulos = JSON.parse(localStorage.getItem('articulos'));
				if (i == 0 || i%2 == 0) {
					articuloNombre = e.target.parentElement.parentElement.parentElement.children[1].children[0].textContent;
					if (JSON.parse(localStorage.getItem("articulos")).find(element => element.nombre == articuloNombre)){
						localArticulos.map(data=>{
							if(articuloNombre == data.nombre){
								if (data.cantidad > 1){
									data.cantidad = data.cantidad - 1;
									localStorage.setItem('articulos', JSON.stringify(localArticulos));
									let cantidadCarrito = document.getElementsByClassName("cantidad");
									cantidadCarrito[i/2].innerHTML = `${data.cantidad}`;
									let cantidadPrecio = document.getElementsByClassName("precio");
									cantidadPrecio[i/2].innerHTML = `${data.precio*data.cantidad}`
								}
								else {
									articuloId = e.target.parentElement.parentElement.parentElement.children[4].children[0].getAttribute('data-id');
									articulos = JSON.parse(localStorage.getItem('articulos'));
									articulos = articulos.filter(articulo => articulo.id != articuloId);
									if(articulos.length == false){ //un array vacio devuelve el booleano false
										localStorage.setItem('articulos', JSON.stringify(articulos));
										vaciarHTML();
										modificarCantidad = document.getElementsByClassName('btn-modificadores');
									}
									else{
										localStorage.setItem('articulos', JSON.stringify(articulos));
										vaciarHTML();
										agregarCarritoHTML();
										modificarCantidad = document.getElementsByClassName('btn-modificadores');
									}
									articulos = [];
									btnVaciar();
									comprobarStorage();
									evitarClick();
								}
							}
						});
					}
				}
				else if (i%2 !== 0){
					articuloNombre = e.target.parentElement.parentElement.parentElement.children[1].children[0].textContent;
					if (JSON.parse(localStorage.getItem("articulos")).find(element => element.nombre == articuloNombre)){
						localArticulos.map(data=>{
						if(articuloNombre == data.nombre){
							data.cantidad = data.cantidad + 1;
							localStorage.setItem('articulos', JSON.stringify(localArticulos));
							let cantidadCarrito = document.getElementsByClassName("cantidad");
							cantidadCarrito[i/2-0.5].innerHTML = `${data.cantidad}`;
							let cantidadPrecio = document.getElementsByClassName("precio");
							cantidadPrecio[i/2-0.5].innerHTML = `${data.precio*data.cantidad}`
						}
							
					});
					}
				}
			});
		
		}
